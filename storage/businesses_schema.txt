CREATE TABLE IF NOT EXISTS "businesses" ("id" integer primary key autoincrement not null, "user_id" integer not null, "business_type_id" integer, "name" varchar not null, "description" text not null, "created_at" datetime, "updated_at" datetime, "business_mode" varchar check ("business_mode" in ('product', 'service', 'both')) not null default 'product', "logo_url" varchar, "established_date" date, "address" text, "employee_count" integer, "revenue_range" varchar, "is_from_college_project" tinyint(1) not null default ('0'), "is_continued_after_graduation" tinyint(1) not null default ('1'), "legal_documents" text, "product_certifications" text, "business_challenges" text, "is_featured" tinyint(1) not null default '0', "position" varchar, "city" varchar, "province" varchar, "additional_data" text, "legal_document_path" varchar, "certification_path" varchar, foreign key("business_type_id") references business_types("id") on delete cascade on update no action, foreign key("user_id") references users("id") on delete cascade on update no action);
CREATE INDEX "businesses_is_featured_index" on "businesses" ("is_featured");
